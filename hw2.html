# x(0) = [x1(0), 0, np.sqrt(x1(0))]
        
# This is the ODE function that should be evaluated
# Can be done in more or less sophisticated ways but this is fine.
def f(_, x):
    dot1 = x[1]
    dot2 = -x[0] - b * x[1] + x[2] * x[2]
    dot3 = -c * (1 - x[0]) * x[2] + u_star
    return [ dot1, dot2, dot3 ]

Exercise 3. State feedback linearization [4p]¶

In [8]:
A = np.array([[0, 1, 0], [-1, -b2, 1], [0, 0, 0]])
B = np.array([[0], [0], [1]])

# These functions are only for ploting v
# Otherwise, they are in some sense obsolete because they are not included in the closed loop dynamics.
def gama(y):
    return (b3 + 2 * y[2])

def pitchfork(y):
    return ((b3 + 2 * y[2]) * (c1 * y[0] + c3 * y[2] + c13 * y[0] * y[2]))

Choose imput signal $v = \frac{1}{\gamma(y)}(-\psi(y) + u) = \frac{1}{\gamma(y)}(-\psi(y) + \textbf{KT}(y))$

so

$\dot{z} = \textbf{A}z + \textbf{B}(\psi(y) + \frac{\gamma(y)}{\gamma(y)}(-\psi(y) + \textbf{K}\underbrace{\textbf{T}(y)}_z) = (\textbf{A} - \textbf{BK})z$

In other words: Fight nonlinearities with magic more nonlinearity! This results in no nonlinearities in the closed loop dynamics.
